###### 更新预发布环境的流程
```text
 cd /data/ubt/adis
 pkill adis  //杀掉服务进程
 rm adis-server  //删掉旧的服务文件
 rz  //上传新的服务文件
 history |grep adis  //查询之前用过的指令
 ps -aux |grep adis  //查看进程
 chmod +x adis-server //赋予权限
 nohup  /data/ubt/adis/adis-server -c /data/ubt/adis/linux.ini  & //在系统后台非挂起运行
 ps -aux |grep adis
```
